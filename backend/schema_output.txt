CREATE TABLE `grades` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `studentId` varchar(255) NOT NULL,
  `courseCode` varchar(255) NOT NULL,
  `courseName` varchar(255) NOT NULL,
  `grade` varchar(255) NOT NULL,
  `score` int(11) NOT NULL,
  `creditHours` int(11) NOT NULL,
  `semester` varchar(255) NOT NULL,
  `uploadedBy` varchar(255) NOT NULL,
  `uploadDate` datetime DEFAULT NULL,
  `published` tinyint(1) DEFAULT 1,
  `parentNotified` tinyint(1) DEFAULT 0,
  `academicYear` varchar(255) DEFAULT '2024',
  `remarks` text DEFAULT NULL,
  `approvalStatus` varchar(255) DEFAULT 'published' COMMENT 'Admin uploads: published, Teacher uploads: pending_approval',
  `approvedBy` int(11) DEFAULT NULL COMMENT 'Admin ID who approved the grade',
  `approvalDate` datetime DEFAULT NULL,
  `rejectionReason` text DEFAULT NULL,
  `submittedDate` datetime DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `studentId` (`studentId`),
  KEY `uploadedBy` (`uploadedBy`),
  CONSTRAINT `grades_ibfk_1` FOREIGN KEY (`studentId`) REFERENCES `students` (`studentId`) ON DELETE NO ACTION ON UPDATE CASCADE,
  CONSTRAINT `grades_ibfk_2` FOREIGN KEY (`uploadedBy`) REFERENCES `teachers` (`teacherId`) ON DELETE NO ACTION ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci
